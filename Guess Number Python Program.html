        else:
            print("Too high!")
            high = min(high, guess - 1)  # Update upper boundary if necessary
        
        # Computer provides a new guess within the updated boundaries
        print(f"Try guessing between {low} and {high}.")

Here’s how you could write a Python program that generates a random, but solvable, 8-puzzle:

python
import random

def is_solvable(puzzle):
    """Check if the puzzle is solvable."""
    # Find the puzzle linearized excluding the empty space
    one_d_puzzle = [num for row in puzzle for num in row if num != 0]
    inversions = 0
    
    # Count inversions
    for i in range(len(one_d_puzzle)):
        for j in range(i + 1, len(one_d_puzzle)):
            if one_d_puzzle[i] > one_d_puzzle[j]:
                inversions += 1

    # The puzzle is solvable if the number of inversions is even
    return inversions % 2 == 0

def generate_puzzle():
    """Generate a random 8-puzzle board."""
    numbers = list(range(9))  # numbers 0-8, where 0 represents the empty space
    random.shuffle(numbers)
    
    # Create a 3x3 grid
    puzzle = [numbers[i*3:(i+1)*3] for i in range(3)]

    # Ensure the puzzle is solvable; reshuffle if it's not
    while not is_solvable(puzzle):
        random.shuffle(numbers)
        puzzle = [numbers[i*3:(i+1)*3] for i in range(3)]

    return puzzle

def print_puzzle(puzzle):
    """Print the puzzle in a readable format."""
    for row in puzzle:
        print(' '.join(['_' if num == 0 else str(num) for num in row]))

# Generate and print a random solvable puzzle
puzzle = generate_puzzle()
print_puzzle(puzzle)

  • Defines generate_puzzle to create a random 8-puzzle. It keeps shuffling the numbers until it forms a solvable puzzle.

Creating an 8-puzzle by starting with a solved board and then making a series of legal moves to shuffle the puzzle is an effective method to ensure the puzzle remains solvable. This approach simulates the sliding of the tiles, ensuring that every generated puzzle state is reachable from the initial solved state.

Here’s a Python program that does just that:

python
import random

def find_empty(puzzle):
    """Find the position of the empty space (0) in the puzzle."""
    for i, row in enumerate(puzzle):
        for j, value in enumerate(row):
            if value == 0:
                return i, j

def move(puzzle, direction):
    """Move the empty space in the specified direction if possible."""
    x, y = find_empty(puzzle)
    moves = {'up': (x - 1, y), 'down': (x + 1, y), 'left': (x, y - 1), 'right': (x, y + 1)}
    
    if direction in moves:
        nx, ny = moves[direction]
        if 0 <= nx < 3 and 0 <= ny < 3:  # Check boundaries
            puzzle[x][y], puzzle[nx][ny] = puzzle[nx][ny], puzzle[x][y]
            return True  # Return True when move is successful
    return False

def shuffle_puzzle(puzzle, num_moves=100):
    """Perform a series of valid moves to shuffle the puzzle."""
    moves = ['up', 'down', 'left', 'right']
    move_history = []
    
    for _ in range(num_moves):
        move_made = False
        while not move_made:
            move_made = move(puzzle, random.choice(moves))
            if move_made:
                move_history.append(moves)  # Store the successful move

    return move_history

def print_puzzle(puzzle):
    """Print the puzzle in a readable format."""
    for row in puzzle:
        print(' '.join(['_' if num == 0 else str(num) for num in row]))
    print()

# Start with a solved puzzle
puzzle = [[1, 2, 3], [4, 5, 6], [7, 8, 0]]
print("Starting from a solved puzzle:")
print_puzzle(puzzle)

# Shuffle the puzzle
history = shuffle_puzzle(puzzle)
print("Shuffled Puzzle:")
print_puzzle(puzzle)
print("Here is the puzzle to solve. Good luck!")

  • Defines move to attempt to slide a tile into the empty space from a specified direction.
  • Defines shuffle_puzzle to make a number of random moves, effectively shuffling the puzzle.
  • Uses print_puzzle to display the puzzle.

Here’s a simple Python program that performs a depth-first search (DFS) to find a solution to any given 8-puzzle state. This approach will use recursion to explore possible moves from a given state until the solution is found.

Below is a simplified implementation that includes tracking the path of moves:

python
def is_goal(puzzle):
    """Check if the puzzle is in the goal state."""
    goal = [[1, 2, 3], [4, 5, 6], [7, 8, 0]]
    return puzzle == goal

def move(puzzle, direction):
    """Attempt to move the empty tile in the specified direction and return the new state."""
    x, y = find_empty(puzzle)
    moves = {'up': (x - 1, y), 'down': (x + 1, y), 'left': (x, y - 1), 'right': (x, y + 1)}
    
    if direction in moves:
        nx, ny = moves[direction]
        if 0 <= nx < 3 and 0 <= ny < 3:
            # Copy the puzzle to avoid modifying the original
            new_puzzle = [row[:] for row in puzzle]
            new_puzzle[x][y], new_puzzle[nx][ny] = new_puzzle[nx][ny], new_puzzle[x][y]
            return new_puzzle
    return None

def dfs(puzzle, path, depth_limit=30):
    """Perform a depth-first search up to a specified depth limit."""
    if is_goal(puzzle):
        return path
    
    if len(path) > depth_limit:
        return None
    
    moves = ['up', 'down', 'left', 'right']
    for move in moves:
        new_puzzle = move(puzzle, move)
        if new_puzzle and (len(path) == 0 or path[-1] != move):
            # Avoid moving back and forth between the same states
            result = dfs(new_puzzle, path + [move], depth_limit)
            if result:
                return result
    return None

def solve_puzzle(puzzle):
    """Solve the puzzle using DFS and return the list of moves."""
    result = dfs(puzzle, [], 30)  # Depth limit can be adjusted
    if result is None:
        return "No solution found within the depth limit."
    else:
        return result

To optimize the solution for the 8-puzzle problem, we can employ the A* search algorithm, which is more efficient than a simple depth-first search. A* uses a heuristic to estimate the cost from the current state to the goal, thereby optimizing the search process to explore the most promising paths first.

For the 8-puzzle, a common heuristic is the Manhattan Distance, which calculates the total number of moves each tile is away from its goal position. Here’s how you can implement the A* search algorithm for the 8-puzzle using the Manhattan Distance heuristic:

python
import heapq

def manhattan_distance(puzzle):
    """Calculate the total Manhattan distance of tiles from their goal positions."""
    goal_position = {1: (0, 0), 2: (0, 1), 3: (0, 2),
                     4: (1, 0), 5: (1, 1), 6: (1, 2),
                     7: (2, 0), 8: (2, 1), 0: (2, 2)}
    distance = 0
    for i in range(3):
        for j in range(3):
            num = puzzle[i][j]
            if num != 0:
                goal_x, goal_y = goal_position[num]
                distance += abs(goal_x - i) + abs(goal_y - j)
    return distance

def a_star_search(puzzle):
    """Perform the A* search to find the solution."""
    # Priority queue for A* search
    open_list = []
    heapq.heappush(open_list, (manhattan_distance(puzzle), 0, puzzle, []))
    visited = set()
    
    while open_list:
        current_cost, current_steps, current_state, path = heapq.heappop(open_list)
        
        if is_goal(current_state):
            return path
        
        if tuple(map(tuple, current_state)) in visited:
            continue
        
        visited.add(tuple(map(tuple, current_state)))
        
        moves = ['up', 'down', 'left', 'right']
        for move in moves:
            new_state = move(current_state, move)
            if new_state and tuple(map(tuple, new_state)) not in visited:
                new_path = path + [move]
                new_steps = current_steps + 1
                heuristic = manhattan_distance(new_state)
                total_cost = heuristic + new_steps
                heapq.heappush(open_list, (total_cost, new_steps, new_state, new_path))
    
    return "No solution found."

# Example of a shuffled puzzle
puzzle = [[1, 2, 3], [4, 0, 6], [7, 5, 8]]
print("Initial Puzzle:")
print_puzzle(puzzle)

solution = a_star_search(puzzle)
if isinstance(solution, list):
    print("Solution found:")
    print(" -> ".join(solution))
else:
    print(solution)

This code:

  • Defines the manhattan_distance function that calculates the Manhattan distance for each tile from its current position to its target position.
  • Implements the a_star_search function using a priority queue managed by heapq to explore the states. It prioritizes states with a lower estimated cost to reach the goal.
  • Uses a set to track visited states to avoid revisiting the same state.

A* search is generally more efficient than depth-first search because it uses heuristics to guide the search towards the goal state, reducing the number of states that need to be explored. This implementation should handle typical 8-puzzle configurations effectively, providing a path to the solution in a reasonable amount of time.
